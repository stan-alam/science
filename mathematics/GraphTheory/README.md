```python
#/scr/graph.py  
graph = { "a" : ["c"],
          "b" : ["c" "d"],
          "c" : ["a", "b", "d"],
          "d" : ["c"]
          }
# letters as nodes
def generate_edges(graph):
    edges = []
    for node in graph:
        for neighbor in graph[node]:
            edges.append((node, neighbor))

    return edges

print(generate_edges(graph))
```
